public class IslamicBanking extends Conventional implements Account {
     
    private String AccountType;
    private double zakat;
    
    IslamicBanking(){}
    IslamicBanking(String name,String fname,String nmbr,String CNIC,String Address,long accountNumber,String profession,String email,String nationality,String income,String soincome,String pass){
        super(name,fname,nmbr,CNIC,Address,accountNumber,profession,email,nationality,income,soincome,pass,"Islamic Banking");
        this.AccountType="Islamic Account";
    }
   
   
    
    
@Override
public void deposit(double amount){
    if(amount > 0){
        this.setBalance(this.getBalance() + amount);
        this.setTransaction(new Transaction("Deposit", amount, this.getBalance()));
        System.out.println("Deposit successful: " + amount);
    } else {
        System.out.println("Please enter a valid amount");
    }
}


    
    public boolean withDraw(double amount){
        if(this.getBalance()>=amount){
        this.setBalance(this.getBalance()-amount);
        this.setTransaction(new Transaction("Withdraw",amount,this.getBalance()));
        return true;}
        else{
            System.out.println("You have insufficant Balance");
            return false;
        }
    }
    
   public boolean transfer(double amount,long number){
        if(this.getBalance()>=amount){
        this.setBalance(this.getBalance()-amount);
        this.setTransaction(new Transaction("Transfer",amount,this.getBalance(),number));
        return true;}
        else{
            System.out.println("You have insufficant Balance");
            return false;
        }
    }
    
    
   public String getAccountDetails(){
        return "Account Number : "+this.getAccountNumber()+"\nAccount Holder : "+this.getAccountHolderName()+"\nAccount Type : Current Account\nBalance : "+this.getBalance();
    }
   
   public void ZakatApply(){
       long now=System.currentTimeMillis();
       //if(this.getBalance()>=200000 && (this.getLastUpdateDate()== null || this.getLastUpdateDate().isBefore(LocalDate.now().minusyear(1)))){    
    if (getLastUpdateDate() == 0L || (now - getLastUpdateDate()) >= 180000) {
   if(this.getBalance() >= 200000){
           this.zakat=this.getBalance()*0.025;
           this.setBalance(this.getBalance()-zakat);
           this.setTransaction(new Transaction("Zakat Deducted",this.zakat,this.getBalance()));
       
       this.setLastUpdateDate(now);}
   }}
    
}

public class generateAccountNumber {
    
    public static long number(){
        long num=0;
        try{
            BufferedReader bis=new BufferedReader(new FileReader("D:\\Semester 02\\OOPS\\BankingManagementSystem\\src\\main\\java\\com\\mycompany\\bankingmanagementsystem\\file\\accountNumber.txt"));
           
            String line=bis.readLine();
            if(line!=null && !line.isEmpty()){
                num=Long.parseLong(line);
                bis.close();
            }}catch(Exception ee){
                System.out.println("Error in "+ee);
                
            }
        num++;
        
        
        try{
            BufferedWriter bw=new BufferedWriter(new FileWriter("D:\\Semester 02\\OOPS\\BankingManagementSystem\\src\\main\\java\\com\\mycompany\\bankingmanagementsystem\\file\\accountNumber.txt"));
            bw.write(String.valueOf(num));
            bw.close();
        }catch(Exception ee){
                System.out.println("Error in "+ee);
                
            }
        return num;
    }
    
    
}

public class Transaction implements Serializable  {
    private String type;
    private double amount;
    private double balance;
    private String dateTime;
    private long reciepentNumber;
    
    Transaction(String type,double amount,double balance){
        this.type=type;
        this.amount=amount;
        this.balance=balance;
        this.dateTime=LocalDateTime.now().toString();
    }
    
    Transaction(String type,double amount,double balance,long num){
        this.type=type;
        this.amount=amount;
        this.balance=balance;
        this.dateTime=LocalDateTime.now().toString();
        this.reciepentNumber=num;
    }

    public long getReciepentNumber(){
        return this.reciepentNumber;
    }
    
    public String getType() {
        return type;
    }

    public double getAmount() {
        return amount;
    }

    public double getBalance() {
        return balance;
    }

    public String getDateTime() {
        return dateTime;
    }
}

    }
}

    
    public boolean withDraw(double amount){

        }
